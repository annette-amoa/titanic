#Classifying all passengers under the age of 16 as children in the gender column
def male_female_child(passenger):
    age,sex = passenger
    
    if age < 16:
        return 'child'
    else:
        return sex
In [26]:
titanic_df['person'] = titanic_df[['Age','Sex']].apply(male_female_child,axis=1)
